127 // Конструктор из вложенного списка инициализации.
128 // Проверяет согласованность размеров строк.
129 Matrix(std::initializer_list<std::initializer_list<T>> init) {
130 const size_type rows = init.size();
131 const size_type cols = rows ? std::begin(init)->size() : 0;
132
133 for (auto const& row : init) {
134 if (row.size() != cols) {
135 throw std::invalid_argument("Matrix initializer list has inconsistent row sizes");
136 }
137 }
138
139 if constexpr (has_static_rows) {
140 if (rows != Rows) {
141 throw std::invalid_argument("Matrix initializer list size does not match static row count");
142 }
143 }
144 if constexpr (has_static_cols) {
145 if (cols != Cols) {
146 throw std::invalid_argument("Matrix initializer list size does not match static column count");
147 }
148 }
149
150 if constexpr (!is_static) {
151 set_size(rows, cols);
152 data_.assign(rows * cols, T{});
153 }
154
155 auto out = data();
156 size_type index = 0;
157 for (auto const& row : init) {
158 for (auto const& value : row) {
159 out[index++] = value;
160 }
161 }
162
163 if constexpr (is_static) {
164 if (index < data_.size()) {
165 std::fill(data_.begin() + index, data_.end(), T{});
166 }
167 }

437 // Матричное умножение. Возвращает новую матрицу с размерностью rows x other.cols().
438 // Для больших матриц используется блочная реализация для лучшей кэш-локальности.
439 template<Arithmetic U, std::size_t R2, std::size_t C2, typename Alloc2>
440 auto operator*(const Matrix<U, R2, C2, Alloc2>& other) const {
441 using result_type = std::common_type_t<T, U>;
442 constexpr size_type out_rows = Rows;
443 constexpr size_type out_cols = C2;
444 using result_matrix = Matrix<result_type, out_rows, out_cols, Allocator>;
445
446 if (cols() != other.rows()) {
447 throw std::invalid_argument("Matrix multiplication dimension mismatch");
448 }
449
450 result_matrix result;
451 if constexpr (!result_matrix::is_static) {
452 result.resize(rows(), other.cols());
453 }
454
455 const size_type r = rows();
456 const size_type common = cols();
457 const size_type c = other.cols();
458
459 if (r > 64 && common > 64 && c > 64) {
460 const size_type block = 32;
461 for (size_type ii = 0; ii < r; ii += block) {
462 for (size_type kk = 0; kk < common; kk += block) {
463 for (size_type jj = 0; jj < c; jj += block) {
464 const size_type iend = std::min(ii + block, r);
465 const size_type kend = std::min(kk + block, common);
466 const size_type jend = std::min(jj + block, c);
467 for (size_type i = ii; i < iend; ++i) {
468 for (size_type k = kk; k < kend; ++k) {
469 const result_type aval = static_cast<result_type>((*this)(i, k));
470 for (size_type j = jj; j < jend; ++j) {
471 result(i, j) += aval * static_cast<result_type>(other(k, j));
472 }
473 }
474 }
475 }
476 }
477 }
478 } else {
479 for (size_type i = 0; i < r; ++i) {
480 for (size_type j = 0; j < c; ++j) {
481 result(i, j) = result_type{};
482 for (size_type k = 0; k < common; ++k) {
483 result(i, j) += static_cast<result_type>((*this)(i, k)) * static_cast<result_type>(other(k, j));
484 }
485 }
486 }
487 }
488 return result;
489 }

518 T determinant() const {
519 if (rows() != cols()) {
520 throw std::logic_error("Determinant is defined only for square matrices");
521 }
522 const size_type n = rows();
523 if (n == 0) {
524 return T{1};
525 }
526 Matrix temp = *this;
527 using calc_type = std::common_type_t<T, double>;
528 calc_type det = 1;
529 int sign = 1;
530
531 for (size_type k = 0; k < n; ++k) {
532 size_type pivot = k;
533 calc_type max_value = detail::abs_value(static_cast<calc_type>(temp(k, k)));
534 for (size_type i = k + 1; i < n; ++i) {
535 const calc_type current = detail::abs_value(static_cast<calc_type>(temp(i, k)));
536 if (current > max_value) {
537 max_value = current;
538 pivot = i;
539 }
540 }
541 if (max_value == calc_type{}) {
542 return T{};
543 }
544 if (pivot != k) {
545 for (size_type j = k; j < n; ++j) {
546 std::swap(temp(k, j), temp(pivot, j));
547 }
548 sign = -sign;
549 }
550 det *= static_cast<calc_type>(temp(k, k));
551 for (size_type i = k + 1; i < n; ++i) {
552 const calc_type factor = static_cast<calc_type>(temp(i, k)) / static_cast<calc_type>(temp(k, k));
553 for (size_type j = k + 1; j < n; ++j) {
554 temp(i, j) -= static_cast<T>(factor * static_cast<calc_type>(temp(k, j)));
555 }
556 }
557 }
558 return static_cast<T>(det * sign);
559 }

561 // Обратная матрица, вычисляется через приведение к расширенной матрице и
562 // гауссово-жорданово исключение. Бросает std::domain_error для вырожденных.
563 Matrix inverse() const {
564 if (rows() != cols()) {
565 throw std::logic_error("Inverse is defined only for square matrices");
566 }
567 const size_type n = rows();
568 if (n == 0) {
569 return Matrix{};
570 }
571 using calc_type = std::common_type_t<T, double>;
572 Matrix<calc_type, Dynamic, Dynamic> augmented(n, 2 * n);
573 for (size_type i = 0; i < n; ++i) {
574 for (size_type j = 0; j < n; ++j) {
575 augmented(i, j) = static_cast<calc_type>((*this)(i, j));
576 }
577 for (size_type j = 0; j < n; ++j) {
578 augmented(i, n + j) = (i == j) ? calc_type{1} : calc_type{0};
579 }
580 }
581
582 for (size_type k = 0; k < n; ++k) {
583 size_type pivot = k;
584 calc_type max_value = detail::abs_value(augmented(k, k));
585 for (size_type i = k + 1; i < n; ++i) {
586 const calc_type current = detail::abs_value(augmented(i, k));
587 if (current > max_value) {
588 max_value = current;
589 pivot = i;
590 }
591 }
592 if (max_value == calc_type{}) {
593 throw std::domain_error("Matrix is singular and cannot be inverted");
594 }
595 if (pivot != k) {
596 for (size_type j = 0; j < 2 * n; ++j) {
597 std::swap(augmented(k, j), augmented(pivot, j));
598 }
599 }
600 const calc_type pivot_value = augmented(k, k);
601 for (size_type j = 0; j < 2 * n; ++j) {
602 augmented(k, j) /= pivot_value;
603 }
604 for (size_type i = 0; i < n; ++i) {
605 if (i == k) {
606 continue;
607 }
608 const calc_type factor = augmented(i, k);
609 for (size_type j = 0; j < 2 * n; ++j) {
610 augmented(i, j) -= factor * augmented(k, j);
611 }
612 }
613 }
614
615 Matrix result;
616 if constexpr (!decltype(result)::is_static) {
617 result.resize(n, n);
618 }
619 for (size_type i = 0; i < n; ++i) {
620 for (size_type j = 0; j < n; ++j) {
621 result(i, j) = static_cast<T>(augmented(i, n + j));
622 }
623 }
624 return result;
625 }

759 /**
760 * Сравнение матриц на равенство.
761 * Для плавающих типов используется сравнение с допустимой погрешностью.
762 */
763 bool operator==(const Matrix& other) const {
764 if (rows() != other.rows() || cols() != other.cols()) {
765 return false;
766 }
767 for (size_type i = 0; i < size(); ++i) {
768 if constexpr (std::is_floating_point_v<T>) {
769 if (detail::abs_value(data()[i] - other.data()[i]) > epsilon) {
770 return false;
771 }
772 } else {
773 if (data()[i] != other.data()[i]) {
774 return false;
775 }
776 }
777 }
778 return true;
779 }

781 /**
782 * Лексикографическое сравнение матриц.
783 * Сначала сравниваются размеры, затем элементы по порядку.
784 */
785 auto operator<=>(const Matrix& other) const {
786 if (rows() != other.rows()) {
787 if (rows() < other.rows()) {
788 return std::strong_ordering::less;
789 }
790 return std::strong_ordering::greater;
791 }
792 if (cols() != other.cols()) {
793 if (cols() < other.cols()) {
794 return std::strong_ordering::less;
795 }
796 return std::strong_ordering::greater;
797 }
798 for (size_type i = 0; i < size(); ++i) {
799 if constexpr (detail::ThreeWayComparable<T>) {
800 auto cmp = data()[i] <=> other.data()[i];
801 if (cmp != 0) {
802 return cmp;
803 }
804 } else {
805 if (data()[i] < other.data()[i]) {
806 return std::strong_ordering::less;
807 }
808 if (data()[i] > other.data()[i]) {
809 return std::strong_ordering::greater;
810 }
811 }
812 }
813 return std::strong_ordering::equal;
814 }

976template<Arithmetic T, std::size_t Rows, std::size_t Cols, typename Alloc>
977std::istream& operator>>(std::istream& is, Matrix<T, Rows, Cols, Alloc>& matrix) {
978 if constexpr (!matrix.is_static) {
979 std::vector<T> values;
980 std::string line;
981 std::size_t cols = 0;
982 std::size_t rows = 0;
983
984 while (std::getline(is, line)) {
985 if (line.empty()) {
986 break;
987 }
988 std::istringstream row_stream(line);
989 T value;
990 std::size_t row_cols = 0;
991 while (row_stream >> value) {
992 values.push_back(value);
993 ++row_cols;
994 }
995 if (row_cols == 0) {
996 continue;
997 }
998 if (cols == 0) {
999 cols = row_cols;
1000 } else if (row_cols != cols) {
1001 is.setstate(std::ios::failbit);
1002 return is;
1003 }
1004 ++rows;
1005 }
1006 matrix.resize(rows, cols);
1007 for (std::size_t i = 0; i < values.size(); ++i) {
1008 matrix.data()[i] = values[i];
1009 }
1010 return is;
1011 } else {
1012 for (std::size_t i = 0; i < matrix.rows(); ++i) {
1013 for (std::size_t j = 0; j < matrix.cols(); ++j) {
1014 if (!(is >> matrix(i, j))) {
1015 return is;
1016 }
1017 }
1018 }
1019 return is;
1020 }
